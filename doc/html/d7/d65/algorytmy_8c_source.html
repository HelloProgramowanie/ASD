93void sortowanie_przez_wstawianie(int* A, int N)
94{
95 int i, j, klucz;
96 for(j=1; j<N; j++){
97 klucz = A[j];
98 // wstawianie A[j] do posortowanej tablicy A[0..j-1]

141void sortowanie_grzebieniowe_grzeb(int* A, int P, int Q, int R)
142{
143 int n1 = Q - P + 1;
144 int n2 = R - Q;
145 int* l=(int*)malloc(sizeof(int)*(n1+1));
146 int* r=(int*)malloc(sizeof(int)*(n2+1));
147 for(int i=0; i<n1; i++){
148 l[i] = A[P+i];
149 }
150 for(int j=0; j<n2; j++){
151 r[j] = A[Q+j+1];
152 }
153 l[n1]=INT_MAX;
154 r[n2]=INT_MAX;
155 int i=0;
156 int j=0;
157 for(int k = P; k<=R; k++){
158 if(l[i]<=r[j]){
159 A[k] = l[i++];
160 } else {
161 A[k] = r[j++];
162 }
